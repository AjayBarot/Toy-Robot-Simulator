def move(position)
  y = position.y
  x = position.x
  f = position.f

  # Coordinates manipulation
  # as per directions
  y += 1 if f.match?('NORTH')
  x += 1 if f.match?('EAST')
  y -= 1 if f.match?('SOUTH')
  x -= 1 if f.match?('WEST')

  Position.new(x, y, f)
end

def place(command)
  _command, x, y, f = command.tr(',', ' ').split

  Position.new(x.to_i, y.to_i, f)
end
